
export PROJECT_ID=c4hnrd-dev
gcloud config set project ${PROJECT_ID}

SA_NAME="sa-job"
SA_FULL_NAME="$SA_NAME@${PROJECT_ID}.iam.gserviceaccount.com"
SA_DESCRIPTION="Service Account for running Jobs"
SA_ROLE="projects/c4hnrd-dev/roles/rolejob"

gcloud iam service-accounts create $SA_NAME \
    --description="$SA_DESCRIPTION" \
    --display-name="$SA_NAME"

gcloud projects add-iam-policy-binding ${PROJECT_ID} \
    --member="serviceAccount:$SA_FULL_NAME" \
    --role="$SA_ROLE"

gcloud iam service-accounts list --filter sa-
gcloud iam roles list --project=${PROJECT_ID}
